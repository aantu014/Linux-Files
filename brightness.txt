##Get graphics card model.
ls /sys/class/backlight/ 

##Allow users in the video group to change the brightness.
vi /etc/udev/rules.d/backlight.rules 

##Add these file for KERNEL enter your graphics model. Mines is intel_backlight. 
ACTION=="add", SUBSYSTEM=="backlight", KERNEL=="intel_backlight", RUN+="/bin/chgrp video /sys/class/backlight/%k/brightness"
ACTION=="add", SUBSYSTEM=="backlight", KERNEL=="intel_backlight", RUN+="/bin/chmod g+w /sys/class/backlight/%k/brightness"

usermod -aG video chino  ##Add user to video group

## Write a script of whatever name you want and save it in your home folder or another location.
##I will name it brightness saved in /home/chino/Scripts/
##Write in script the following:

#!/bin/bash

echo "Enter bightness level"
read lvl
int="^[0-9]+$"

if ! [[ $lvl =~ $int ]] ; then
    echo "error: Not a number" >&2; exit 1
else
    echo $lvl | tee /sys/class/backlight/intel_backlight/brightness
fi

##Remember I'm using intel_backlight. If different graphic card model change it.

##Change permission of file to allow it to execute.
chmod 744 brightness

##Now edit .bashrc which is in your home root
vi /home/chino/.bashrc

## This will allow you to execute the file as a command called "bri"
## You can change the name if you want
brightness="/home/chino/Scripts/brightness"
alias bri="$brightness"
